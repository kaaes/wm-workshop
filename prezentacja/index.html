  evt.target;           // the element that triggered the event
  evt.currentTarget;    // the element that listener was added to

  evt.clientX;          // x coordinate of the clicked point
  evt.clientY;          // y coordinate of the clicked point
  evt.button;           // left or right button was used?

  /* KeyboardEvents */

  evt.charCode;        // the Unicode character value of the key pressed

    console.log('A link, with the href:' + target.href);
